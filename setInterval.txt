

import React from "react";

class App extends React.Component {
  constructor(props) {
    super(props);
    this.state = { txt: "txt 1" };
  }
  componentDidMount() {
    setInterval(() => {
      this.setState({ txt: "txt 2" });
    }, 3000);
  }
  render() {
    return <h4>this is {this.state.txt}</h4>;
  }
}

export default App;
